#!/bin/bash

# Script 5: Open Source Manifesto Generator

echo "Answer the following questions:"
echo ""

# User input
read -p "1. Open-source tool you use: " TOOL
read -p "2. Meaning of freedom (one word): " FREEDOM
read -p "3. What will you build: " BUILD

DATE=$(date)
OUTPUT="manifesto.txt"

# Create manifesto
echo "" > $OUTPUT
echo "----- Open Source Manifesto -----" >> $OUTPUT
echo "Date: $DATE" >> $OUTPUT
echo "" >> $OUTPUT
echo "I use $TOOL in my daily life." >> $OUTPUT
echo "For me, freedom means $FREEDOM." >> $OUTPUT
echo "I want to build $BUILD and share it with everyone." >> $OUTPUT
echo "I believe in open source and collaboration." >> $OUTPUT

# Display output
echo ""
echo "Manifesto saved in $OUTPUT"
echo ""
cat $OUTPUT
