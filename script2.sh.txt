#!/bin/bash

# Script 2: FOSS Package Inspector

PACKAGE="git"

echo "Checking package: $PACKAGE"
echo "--------------------------"

# Check if git command exists
if command -v git >/dev/null 2>&1; then
    echo "$PACKAGE is installed."

    # Show version
    VERSION=$(git --version)
    echo "Version: $VERSION"

else
    echo "$PACKAGE is NOT installed."
fi

# Case statement for description
echo ""
echo "Package Description:"

case $PACKAGE in
    git)
        echo "Git is a version control system used to track code changes."
        ;;
    apache)
        echo "Apache is a web server used to host websites."
        ;;
    mysql)
        echo "MySQL is a database system used to store data."
        ;;
    firefox)
        echo "Firefox is an open-source web browser."
        ;;
    *)
        echo "Unknown package."
        ;;
esac
